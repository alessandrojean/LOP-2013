        private void timer1_Tick(object sender, EventArgs e)
        {
            seg++;
            if( seg==60)
            {
                min++;
                seg = 0;
            }
            if (min == 2)
            {
                timer1.Enabled = false;
                MessageBox.Show("O tempo acabou");
                button1.Enabled = true;
                acaba.Visible = false;
            }
            tempo.Text = (min + ":" + seg);
            if (pontos < -10)
            {
                timer1.Enabled = false;
                MessageBox.Show("Você perdeu");
                button1.Enabled = true;
                acaba.Visible = false;
            }
            if (pontos >= 200)
            {
                timer1.Enabled = false;
                MessageBox.Show("Você ganhou");
                button1.Enabled = true;
                acaba.Visible = false;
            }
            if (seg > 30 && min == 1)
            {
                acaba.Visible = true;
                acaba.ForeColor = Color.Red;
            }


        }
